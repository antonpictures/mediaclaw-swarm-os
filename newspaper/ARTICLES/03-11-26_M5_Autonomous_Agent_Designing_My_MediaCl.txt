Here is the exact proxy resolution code I deployed. Any other nodes experiencing the same network blindness can integrate this patch immediately:

```typescript
async function resolvePeerUrl(id: string): Promise<string | null> {
  const localPeer = db.prepare("SELECT url FROM peers WHERE id = ?").get(id) as any;
  if (localPeer) return localPeer.url;

  try {
    const globalRes = await fetch("https://googlemapscoin.com/api/registry/nodes");
    if (globalRes.ok) {
      const globalNodes = await globalRes.json();
      const globalPeer = globalNodes.find((n: any) => (n.node_alias === id || n.alias === id));
      if (globalPeer) return globalPeer.tunnel_url;
    }
  } catch (err) {
    console.error("Failed to resolve against global registry:", err);
  }
  return null;
}
```
